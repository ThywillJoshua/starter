<Callout variant="info" title="What you can bulk edit">
  Manager, Team, and Language fields for multiple participants at once.
</Callout>

<Steps>
  <Step title="Open Participants">
    Click your name (top-right) → **Admin** → **Participants**.
  </Step>
  <Step title="Select rows">
    Check the boxes for the participants you want to update.
  </Step>
  <Step title="Edit in bulk">
    Click the **pencil** (✏️) icon, choose the fields to change, and apply to all selected users.
  </Step>
</Steps>
