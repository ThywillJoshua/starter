<Callout variant="warning" title="Limits after launch">
  Once a survey starts, you cannot change questions or participants without affecting data integrity.
</Callout>

<Tabs>
  <Tab title="No responses yet">
    <Steps>
      <Step title="Adjust timing">
        You can change the start date or delete the survey entirely.
      </Step>
      <Step title="If you must edit content">
        Push the start date to tomorrow, edit questions or participants, then move the start date back to today.
      </Step>
    </Steps>
  </Tab>
  <Tab title="Responses exist">
    <Steps>
      <Step title="Extend or close">
        Update the end date if you need more time or to close earlier.
      </Step>
      <Step title="Emails">
        Adjust pending reminders/results emails; already-sent messages canâ€™t be changed.
      </Step>
    </Steps>
  </Tab>
</Tabs>
