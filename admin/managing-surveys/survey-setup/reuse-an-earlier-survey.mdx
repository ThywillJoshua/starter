<Callout variant="info" title="When to reuse">
  Start from a past survey to keep question sets and targeting consistent, then update timing and audience as needed.
</Callout>

<Steps>
  <Step title="Open Surveys">
    Log in as an admin → **Admin** → **Surveys**.
  </Step>
  <Step title="Select the source survey">
    Search for the survey to copy and click the **Reuse** icon on the right.
  </Step>
  <Step title="Update details">
    Set new start/end dates, adjust teams or participants, and tweak questions if needed.
  </Step>
  <Step title="Review emails">
    Confirm invitations/reminders align to the new dates before saving.
  </Step>
</Steps>
